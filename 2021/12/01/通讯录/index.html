通讯录功能函数的头文件

function.h

#pragma once
#include<iostream>
#include<string>
#define MAX 100
using namespace std;

//定义联系人结构体
struct Person
{
    string m_Name;//姓名
    int m_Sex;//性别 1代表男 2代表女
    int m_Age;//年龄
    string m_Phone;//电话
    string m_Addr;//家庭住址
};

//定义通讯录结构体
struct addressBooks
{
    struct Person personArr[MAX];
    int m_Size;
};

//显示菜单
void showMenu();

//输入某一个联系人的信息
void inputPerson(struct Person* person);

//输出某一个联系人的信息
void outputPerson(struct Person* person);

//1、添加联系人
void addPerson(struct addressBooks* abs);

//2、显示联系人
void showPerson(struct addressBooks* abs);

//3、删除联系人
void deletePerson(struct addressBooks* abs);

//4、查找联系人
void searchPerson(struct addressBooks* abs);

//5、修改联系人
void recomposePerson(struct addressBooks* abs);

//6 清空联系人
void clearAddressBooks(struct addressBooks* abs);

通讯录功能函数

function.cpp

#include "function.h"

//显示菜单栏
void showMenu()
{
    cout << "*********通讯录********" << endl;
    cout << "*****1、添加联系人*****" << endl;
    cout << "*****2、显示联系人*****" << endl;
    cout << "*****3、删除联系人*****" << endl;
    cout << "*****4、查找联系人*****" << endl;
    cout << "*****5、修改联系人*****" << endl;
    cout << "*****6、清空联系人*****" << endl;
    cout << "*****0、退出通讯录*****" << endl;
}

//1、添加联系人
void addPerson(struct addressBooks* abs)
{
    if (abs->m_Size == MAX)
    {
        cout << "当前通讯录已满" << endl;
    }
    else
    {
        inputPerson(&(abs->personArr[abs->m_Size]));
        abs->m_Size++;
        cout << "添加成功" << endl;
    }
    system("pause");
    system("cls");
}

//查找某一联系人是否存在
int isExist(struct addressBooks* abs,string name)
{
    int ret = -1;
    for (int i = 0; i < abs->m_Size; i++)
    {
        if (abs->personArr[i].m_Name == name)
        {
            ret = i;
            break;
        }
    }
    return ret;
}

//输入某一联系人的信息
void inputPerson(struct Person* person)
{
    cout << "请输入联系人姓名：" << endl;
    string name;
    cin >> name;
    person->m_Name = name;

    cout << "请输入联系人年龄" << endl;
    int age;
    cin >> age;
    person->m_Age = age;

    cout << "请输入联系人性别" << endl;
    cout << "男---1" << "\t" << "女---2" << endl;
    int sex;
    while (true)
    {
        cin >> sex;
        if (sex == 1 || sex == 2)
        {
            person->m_Sex = sex;
            break;
        }
        cout << "输入有误，请重新输入" << endl;
    }

    cout << "请输入联系人电话" << endl;
    string phone;
    cin >> phone;
    person->m_Phone = phone;

    cout << "请输入联系人家庭地址" << endl;
    string address;
    cin >> address;
    person->m_Addr = address;
}

//输出某一联系人的信息
void outputPerson(struct Person* person)
{
    cout << "姓名：" << person->m_Name << endl;;
    cout << "性别：" << ((person->m_Sex == 1) ? "男" : "女") << "\t";
    cout << "年龄：" << person->m_Age << "\t";
    cout << "联系电话：" << person->m_Phone << "\t";
    cout << "家庭住址：" << person->m_Addr << endl;
}

//2、显示联系人
void showPerson(struct addressBooks* abs)
{
    if (abs->m_Size == 0)
    {
        cout << "当前通讯录为空" << endl;
    }
    else
    {
        for (int i = 0; i < abs->m_Size; i++)
        {
            outputPerson(&(abs->personArr[i]));
        }
    }
    system("pause");
    system("cls");
}

//3、删除联系人
void deletePerson(struct addressBooks* abs)
{
    cout << "请输入要删除的联系人姓名" << endl;
    string name;
    cin >> name;
    int ret = isExist(abs, name);
    if (ret == -1)
    {
        cout << "此联系人不存在" << endl;
    }
    else
    {
        for (int i = ret; i < abs->m_Size - 1; i++)
        {
            abs->personArr[i] = abs->personArr[i + 1];
        }
        abs->m_Size--;
        cout << "删除成功" << endl;
    }
    system("pause");
    system("cls");
}

//5、修改联系人
void recomposePerson(struct addressBooks* abs)
{
    cout << "请输入要修改的联系人姓名" << endl;
    string name;
    cin >> name;
    int ret = isExist(abs, name);
    if (ret == -1)
    {
        cout << "此联系人不存在" << endl;
    }
    else
    {
        inputPerson(&(abs->personArr[ret]));
        cout << "修改成功" << endl;
    }
    system("pause");
    system("cls");
}

//4、查找联系人
void searchPerson(struct addressBooks* abs)
{
    cout << "请输入要查找的联系人姓名" << endl;
    string name;
    cin >> name;
    int ret = isExist(abs, name);
    if (ret == -1)
    {
        cout << "此联系人不存在" << endl;
    }
    else
    {
        outputPerson(&(abs->personArr[ret]));
    }
    system("pause");
    system("cls");
}

//6、清空联系人
void clearAddressBooks(struct addressBooks* abs)
{
    cout << "您确定要删除所有联系人吗" << endl;
    cout << "确定请输入Yes" << endl;
    cout << "取消请输入No" << endl;
    string inputUser;
    while (true)
    {
        cin >> inputUser;
        if (inputUser == "Yes")
        {
            abs->m_Size = 0;
            cout << "清空成功" << endl;
            break;
        }
        else if (inputUser == "No")
        {
            cout << "取消成功" << endl;
            break;
        }
        cout << "输入有误，请重新输入" << endl;
    }
    system("pause");
    system("cls");
}
