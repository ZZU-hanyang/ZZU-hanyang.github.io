结构体

结构体的定义

struct 结构体名
{
    数据类型 数据名；
        数据类型 数据名；
        数据类型 数据名
}；

创建结构体数组

struct 结构体名 数组名[元素个数]=
{
    {},
    {},
        {},
        {}
};

示例

#include<iostream>
#include<string> 
using namespace std;

//结构体的定义
struct student 
{
    string name;
    int age;
    int score;
}; 

int main()
{    

//结构体的创建
    struct student stuArray[4]=
    {
        {"刘备",18,97},
        {"关羽",20,90},
        {"张飞",19,99},
        {"赵云",21,98}
    };

//结构体元素的赋值
    stuArray[2].name="诸葛亮";
    stuArray[2].age=99;
    stuArray[2].score=100;


//结构体的遍历访问
    for(int i=0;i<4;i++)
    {
        cout<<stuArray[i].name<<": "
        <<stuArray[i].age<<"岁 "
        <<stuArray[i].score<<"分"<<endl; 
    }
    system("pause");
    return 0;
}

注意：定义或者创建完结构体之后，末尾要加分号；

在定义完成结构体之后，后续的结构体在创建和赋值时可以省略关键字struct
