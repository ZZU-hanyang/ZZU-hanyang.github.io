运算符

运算符

加法 +

//加法 +
int num1 = 10, num2 = 3;
cout << num1 + num2 << endl;

减法 -

//减法
cout << num1 - num2 << endl;

乘法 *

//乘法
cout << num1*num2 << endl;

除法 /

//除法
cout << num1 / num2 << endl;//两个整数相除的运算结果只保留整数位

/*被除数不能为0，否则会报错
int num3 = 0;
cout << num1 / num3 << endl;*/

//两个小数相除
double num3 = 3.0;
cout << num1/num3 << endl;

取余 %

//取余
cout << num1%num2 << endl;

前置递增

// 前置递增
int a1 = 10;
++a1;
cout <<"a1=" << a1 << endl;

后置递增

//后置递增
int b1 = 10;
b1++;
cout <<"b1=" << b1 << endl;

前置和后置的区别

/*前置递增是先让操作数加1，再进行表达式的运算；后置递增是先进行表达数的运算，再将其加1 */
int a2 = 5, b2 = 5;
cout << "a2=" << ++a2 << endl;//前置
cout << "b2=" << b2++ << endl;//后置

前置和后置递减同理
