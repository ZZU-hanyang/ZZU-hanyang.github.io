二维数组

二维数组的定义

//定义一个2行3列的二维数组 
int arr1[2][3]; 
 
int arr2[2][3]=
{
    {1,2,3},
    {4,5,6}
}; 

int arr3[][3]={{1,2,3},{4,5,6}};

int arr4[2][3]={1,2,3,4,5,6};

int arr5[][3]={1,2,3,4,5,6};

/*
以上定义的都是同一个二维数组，arr4和arr5中，计算机会根据给出的列数
自动计算出数组的行数 
*/

二维数组的访问

//访问一个2行3列的数组
int i,j;
for(i=0;i<2;i++)
{
    for(j=0;j<3;j++)
    {
        cout<<arr2[i][j]<<" ";
    }
    cout<<endl;
 }

二维数组名

int arr[2][3]=
{
    {1,2,3},
    {4,5,6}
};

cout<<"二维数组的首地址是："<<arr<<endl;

cout<<"二维数组占用的内存空间为："<<sizeof(arr)<<endl;

cout<<"二维数组含有的元素个数为："<<sizeof(arr)/sizeof(arr[0][0])<<endl;

cout<<"二维数组第一行的地址为："<<arr[0]<<endl;

cout<<"二维数组第二行第二列的地址为："<<&arr[0][1]<<endl;

cout<<"二维数组第二行的地址为："<<arr[1]<<endl;

cout<<"二维数组第二行的首地址是："<<&arr[1][0]<<endl;

cout<<"二维数组每个元素占用的内存空间为："<<sizeof(arr[0][0])<<endl;

cout<<"二维数组的行数为："<<sizeof(arr)/sizeof(arr[0])<<endl;

cout<<"二维数组的列数为："<<sizeof(arr[0])/sizeof(arr[0][0])<<endl;
